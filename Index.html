Google Script Installation Guide

Implement the secure REST translator in your target Google Sheet Workbook

1

Generate Your Base Spreadsheet

Open a blank document in Google Sheets. Create 5 individual sheets/tabs inside, named exactly: Sales, Warranty, Debtors, StockCount, and Overheads.

2

Open Apps Script Workspace

Go to the navigation menu at the top of your sheet and select Extensions > Apps Script.

3

Paste Script Translator

Overwrite everything in the editor box with the code below:

/* COPY THIS CODE TO GOOGLE APPS SCRIPT */

function doGet() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const data = {};
  const sheets = ["Sales", "Warranty", "Debtors", "StockCount", "Overheads"];
  
  sheets.forEach(name => {
    let sheet = ss.getSheetByName(name);
    if (!sheet) sheet = ss.insertSheet(name);
    
    const values = sheet.getDataRange().getValues();
    if (values.length <= 1) {
      data[name] = [];
    } else {
      const headers = values[0];
      data[name] = values.slice(1).map(row => {
        const obj = {};
        headers.forEach((h, i) => {
          obj[h] = row[i];
        });
        return obj;
      });
    }
  });
  
  return ContentService.createTextOutput(JSON.stringify({ status: "success", data }))
                       .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    const payload = JSON.parse(e.postData.contents);
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    
    for (const sheetName in payload) {
      let sheet = ss.getSheetByName(sheetName);
      if (!sheet) sheet = ss.insertSheet(sheetName);
      
      sheet.clear();
      const records = payload[sheetName];
      if (records.length > 0) {
        const headers = Object.keys(records[0]);
        sheet.appendRow(headers);
        const rows = records.map(rec => headers.map(h => rec[h] !== undefined ? rec[h] : ""));
        sheet.getRange(2, 1, rows.length, headers.length).setValues(rows);
      }
    }
    return ContentService.createTextOutput(JSON.stringify({ status: "success" }))
                         .setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({ status: "error", message: err.toString() }))
                         .setMimeType(ContentService.MimeType.JSON);
  }
}
4

Deploy as a Web App

1. Click Deploy > New Deployment.
2. Under Select Type, choose Web App.
3. Set Execute As to "Me (your-email@gmail.com)".
4. Set Who Has Access to "Anyone" (Required for secure access from this Dashboard).
5. Authorize permissions when prompted by Google, and copy the generated Web App URL into the configuration box on the left!
